19  import org.apache.maven.plugins.annotations.Mojo;
20  import org.apache.maven.plugins.annotations.Parameter;
21  import org.codehaus.mojo.keytool.requests.KeyToolListRequest;
22  
23  /**
24   * To list entries in a keystore.
25   * Implemented as a wrapper around the SDK {@code keytool -list} (jdk 1.5) command.
26   * See <a href="http://java.sun.com/j2se/1.5.0/docs/tooldocs/windows/keytool.html">keystore documentation</a>.
27   *
28   * @author tchemit
29   * @since 1.2
30   */
31  @Mojo( name = "list", requiresProject = true, threadSafe = true )
32  public class ListMojo
33      extends AbstractKeyToolRequestWithKeyStoreAndAliasParametersMojo<KeyToolListRequest>
34  {
35      /**
36       * Output in RFC style.
37       * See <a href="http://docs.oracle.com/javase/1.5.0/docs/tooldocs/windows/keytool.html#Commands">options</a>.
38       *
39       * @since 1.2
40       */
41      @Parameter
42      private boolean rfc;
43  
44      /**
45       * Default contructor.
46       */
47      public ListMojo()
48      {
49          super( KeyToolListRequest.class );
50      }
51  
52      /** {@inheritDoc} */
53      @Override
54      protected KeyToolListRequest createKeytoolRequest()
55      {
56          KeyToolListRequest request = super.createKeytoolRequest();
57  
58          request.setRfc( this.rfc );
59          return request;
60      }
61  
62  }
