19  import org.apache.maven.shared.utils.cli.Commandline;
20  import org.codehaus.plexus.logging.Logger;
21  
22  /**
23   * To build the command line for a given {@link org.codehaus.mojo.keytool.KeyToolRequest}.
24   *
25   * @author tchemit
26   * @since 1.1
27   */
28  public interface KeyToolCommandLineBuilder
29  {
30      /**
31       * Plexus role name.
32       */
33      String ROLE = KeyToolCommandLineBuilder.class.getName();
34  
35      /**
36       * Test if given request type is supported by the underlined keytool implementation.
37       * <strong>Note:</strong> a request of a none supported type will then thrown a {@link org.codehaus.mojo.keytool.UnsupportedKeyToolRequestException} in method {@link #build(KeyToolRequest)}
38       *
39       * @param requestType type of request to test
40       * @param <R>         type of request to test
41       * @return {@code true} if can create a such request type, {@code false} otherwise.
42       * @since 1.3
43       */
44      <R extends KeyToolRequest> boolean supportRequestType( Class<R> requestType );
45  
46      /**
47       * Build the commandline given the incoming keytool request.
48       *
49       * @param request keytool request
50       * @return the prepared commandline client ready to be executed
51       * @throws org.codehaus.mojo.keytool.CommandLineConfigurationException
52       *          if could not find keytool executable
53       * @throws org.codehaus.mojo.keytool.UnsupportedKeyToolRequestException if any.
54       */
55      Commandline build( KeyToolRequest request )
56          throws CommandLineConfigurationException, UnsupportedKeyToolRequestException;
57  
58      /**
59       * Sets the logger used by the builder.
60       *
61       * @param logger logger to use in this builder
62       */
63      void setLogger( Logger logger );
64  
65      /**
66       * Sets the keytool executable location.
67       *
68       * @param keyToolFile keytool executable location to use in this builder
69       */
70      void setKeyToolFile( String keyToolFile );
71  
72      /**
73       * Checks that builder is ready to produce commandline from incoming request.
74       * Says a logger is set and a keytool executable location is setted.
75       */
76      void checkRequiredState();
77  
78      /**
79       * Checks that builder can build the given type of request.
80       *
81       * @param request request to test
82       * @since 1.3
83       * @throws org.codehaus.mojo.keytool.UnsupportedKeyToolRequestException if any.
84       */
85      void checkSupportedRequest( KeyToolRequest request )
86          throws UnsupportedKeyToolRequestException;
87  
88  }
