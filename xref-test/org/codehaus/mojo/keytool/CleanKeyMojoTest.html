19  import junit.framework.TestCase;
20  import org.apache.maven.plugin.MojoExecutionException;
21  import org.apache.maven.plugin.MojoFailureException;
22  
23  import java.io.File;
24  import java.io.IOException;
25  
26  /**
27   * Unit Test class that test the removal of an existing and non existing
28   * keystore file.
29   *
30   * @author Sharmarke Aden (<a href="mailto:saden1@gmail.com">saden</a>)
31   * @author $Author$
32   * @version $Revision$
33   */
34  public class CleanKeyMojoTest
35      extends TestCase
36  {
37  
38      private static final String EXISTING_TEST_KEYSTORE = ".keystore";
39  
40      private static final String NON_EXISTING_TEST_KEYSTORE = ".keystoreZXCVBNNM";
41  
42      private CleanKeyMojo mojo;
43  
44      protected void setUp()
45          throws Exception
46      {
47          mojo = new CleanKeyMojo();
48  
49      }
50  
51      public void tearDown()
52      {
53          mojo = null;
54      }
55  
56      /**
57       * Remove the temp keystore file created in the setup.
58       *
59       * @throws MojoExecutionException
60       * @throws MojoFailureException
61       * @throws IOException
62       */
63      public void testExistingRemoveKeystore()
64          throws MojoExecutionException, MojoFailureException, IOException
65      {
66          String tempKeystore =
67              System.getProperty( "java.io.tmpdir" ) + File.separator + CleanKeyMojoTest.EXISTING_TEST_KEYSTORE;
68          File tempKeystoreFile = new File( tempKeystore );
69          tempKeystoreFile.createNewFile();
70          mojo.setKeystore( tempKeystore );
71          mojo.execute();
72  
73      }
74  
75      /**
76       * Remove the temp keystore file created in the setup. existent
77       *
78       * @throws MojoExecutionException
79       * @throws MojoFailureException
80       */
81      public void testRemoveNonExistentKeystore()
82          throws MojoExecutionException, MojoFailureException
83      {
84          mojo.setKeystore( CleanKeyMojoTest.NON_EXISTING_TEST_KEYSTORE );
85          mojo.execute();
86      }
87  
88  }
